(2) 在pinyougou-sellergoods-service的TypeTemplateServiceImpl.java新增方法

/**************
 *    构建数据格式
 *   [
 *       {"id":1,"text":"内存大小","options":[{"id":123,"optionName":"4G"}]},
 *       {"id":2,"text":"网络制式","options":[{"id":123,"optionName":"移动4G"}]},
 *       {"id":3,"text":"屏幕尺寸","options":[{"id":123,"optionName":"16寸"},{"id":124,"optionName":"60寸"}]}
 *   ];
********/
@Override
public List<Map> getSpecificationOptionById(long id) {
    //先根据模板ID查询出规格列表，规格列表是一个集合[字符串]
    TypeTemplate typeTemplate = typeTemplateMapper.selectByPrimaryKey(id);

    //通过FastJSON把字符串集合转换成集合数据
    List<Map> listMap = JSON.parseArray(typeTemplate.getSpecIds(),Map.class);

    for (Map map : listMap) {
        //循环集合，根据规格的ID查询出规格选项
        long specId = Long.parseLong(map.get("id").toString()); //规格ID
        List<SpecificationOption> options = specificationOptionMapper.select(new SpecificationOption(specId));
        //构建JSON数据options结构
        map.put("options",options);
    }
    return listMap;
}

$scope.updateSpecAttribute=function ($event,name,value) {
    //1、查找到当前规格名字对应的对象
    var object = $scope.searchObjectByKey($scope.entity.goodsDesc.specificationItems,"attributeName",name);
    $log.debug(object);
    if(object!=null){
        if($event.target.checked){
            //执行勾选时
            //2、如果存在，则往它属性里追加当前选中的规格选项
            object.attributeValue.push(value);
        }else{
            //2.1、如果取消，则将该规格选项从attributeValue中移除
            object.attributeValue.splice(object.attributeValue.indexOf(value),1);

            //如果attributeValue的个数为0，则将该规格移除
            if(object.attributeValue.length==0){
                $scope.entity.goodsDesc.specificationItems.splice($scope.entity.goodsDesc.specificationItems.indexOf(object),1);
            }
        }
    }else{
        //3、不存在则，创建一个当前规格，并往它属性里追加当前选中的规格选项
        //[{"attributeName":"网络制式","attributeValue":["移动3G","移动4G"]},{"attributeName":"屏幕尺寸","attributeValue":["6寸","5.5寸"]}]
        $scope.entity.goodsDesc.specificationItems.push({"attributeName":name,"attributeValue":[value]});
    }
}

//创建一条item
$scope.createItemList=function () {
    //单个商品
    var item={spec:{},price:0,num:1,status:'0',isDefault:'0'};
    //集合商品
    $scope.entity.items=[item];
    //重新定义一个变量，写法更简洁
    var speclist = $scope.entity.goodsDesc.specificationItems;
    //循环重组
    for(var i=0;i<speclist.length;i++){
        $scope.entity.items=addColumn($scope.entity.items,speclist[i].attributeName,speclist[i].attributeValue);
    }
}

//商品个数重组
addColumn=function (list,columnName,columnValues) {
    //定义一个新的集合
    var newList=[];
    //原来商品个数循环
    for(var i=0;i<list.length;i++){
        //重组
        for(var j=0;j<columnValues.length;j++){
            //深克隆
            var newItem = angular.copy(list[i]);
            //给新增的规格赋值
            newItem.spec[columnName]=columnValues[j];
            //加入到新的集合中去
            newList.push(newItem);
        }
    }
    return newList;
}

@Override
public int add(Goods goods) {
    //默认状态
    goods.setAuditStatus("0");
    //增加Goods
    int acount = goodsMapper.insertSelective(goods);

    //GoodsDesc
    GoodsDesc goodsDesc = goods.getGoodsDesc();
    goodsDesc.setGoodsId(goods.getId());
    goodsDescMapper.insertSelective(goodsDesc);

    //获取品牌信息
    Brand brand = brandMapper.selectByPrimaryKey(goods.getBrandId());
    //获取商家信息
    Seller seller = sellerMapper.selectByPrimaryKey(goods.getSellerId());
    //商品分类-三级分类
    ItemCat itemCat = itemCatMapper.selectByPrimaryKey(goods.getCategory3Id());

    //当前时间
    Date now = new Date();
    //增加SKU信息
    for (Item item : goods.getItems()) {
        //组装title
        String title = goods.getGoodsName();
        //将Spec（规格转成Map集合）
        Map<String,String> specMap = (Map<String, String>) JSON.parse(item.getSpec());
        for (Map.Entry<String, String> entity : specMap.entrySet()) {
            //拼接标题
            title+=" "+entity.getValue();
        }
        item.setTitle(title);
        //补全其他数据
        item.setGoodsId(goods.getId());     //商品SPU编号
        item.setSellerId(goods.getSellerId());//商家编号
        item.setCategoryid(goods.getCategory3Id());//商品分类编号：三级编号
        item.setCreateTime(now);        //创建日期
        item.setUpdateTime(now);        //修改日期
        //设置品牌信息
        item.setBrand(brand.getName());
        //设置商家信息
        item.setSeller(seller.getNickName());
        //设置三级分类
        item.setCategory(itemCat.getName());
        //获取商品第一张图片
        List<Map> imageMap =JSON.parseArray(goods.getGoodsDesc().getItemImages(),Map.class);
        if(imageMap.size()>0){
            item.setImage(imageMap.get(0).get("url").toString());
        }
        //增加商品SKU入库
        itemMapper.insertSelective(item);
    }
    return acount;
}

/***
 * 增加Goods信息
 * @param goods
 * @return
 */
@Override
public int add(Goods goods) throws Exception{
    //状态设置
    goods.setAuditStatus("0");
    //新增商品
    int acount = goodsMapper.insertSelective(goods);

    //商品详情
    goods.getGoodsDesc().setGoodsId(goods.getId()); //设置主键为商品主键
    goodsDescMapper.insertSelective(goods.getGoodsDesc());

    //获取品牌信息
    Brand brand = brandMapper.selectByPrimaryKey(goods.getBrandId());
    //获取商家信息
    Seller seller = sellerMapper.selectByPrimaryKey(goods.getSellerId());
    //商品分类-三级分类
    ItemCat itemCat = itemCatMapper.selectByPrimaryKey(goods.getCategory3Id());

    //添加SKU
    if(goods.getIsEnableSpec()!=null && goods.getIsEnableSpec().equals("1") && goods.getItems()!=null && goods.getItems().size()>0){
        //循环添加
        for (Item item : goods.getItems()) {
            //组装title
            String title = goods.getGoodsName();
            //将Spec（规格转成Map集合）
            Map<String,String> specMap =objectMapper.readValue(item.getSpec(),Map.class);
            for (Map.Entry<String, String> entity : specMap.entrySet()) {
                //拼接标题
                title+=" "+entity.getValue();
            }
            item.setTitle(title);

            //其他属性赋值
            setItemInfo(item,goods,brand,seller,itemCat);

            //增加商品SKU入库
            itemMapper.insertSelective(item);
        }
    }else{
        //不启用规格时，只增加一个商品信息
        //这里只添加一个商品的SKU
        Item item = new Item();

        //商品KPU+规格描述串作为SKU名称
        item.setTitle(goods.getGoodsName());
        //价格
        item.setPrice(goods.getPrice());
        //状态
        item.setStatus("1");
        //是否默认
        item.setIsDefault("1");
        //库存数量
        item.setNum(9999);
        //规格
        item.setSpec("{}");
        //增加商品SKU入库
        itemMapper.insertSelective(item);
    }
    return acount;
}
/***
 * 设置SKU其他信息
 * @param item
 * @param goods
 * @throws Exception
 */
public void setItemInfo(Item item,Goods goods,Brand brand,Seller seller,ItemCat itemCat) throws Exception{
    //当前时间
    Date now = new Date();

    //补全其他数据
    item.setGoodsId(goods.getId());     //商品SPU编号
    item.setSellerId(goods.getSellerId());//商家编号
    item.setCategoryid(goods.getCategory3Id());//商品分类编号：三级编号
    item.setCreateTime(now);        //创建日期
    item.setUpdateTime(now);        //修改日期

    //设置品牌信息
    item.setBrand(brand.getName());
    //设置商家信息
    item.setSeller(seller.getNickName());
    //设置三级分类
    item.setCategory(itemCat.getName());
    //获取商品第一张图片
    List<Map> imageMap =JSON.parseArray(goods.getGoodsDesc().getItemImages(),Map.class);
    if(imageMap.size()>0){
        item.setImage(imageMap.get(0).get("url").toString());
    }
}
